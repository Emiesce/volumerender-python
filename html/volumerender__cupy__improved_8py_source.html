22def transferFunction(x):
23 # Apply exponential functions to simulate different material densities
24 calculation_1 = cp.exp(-(x - 9.0) ** 2 / 1.0)
25 calculation_2 = cp.exp(-(x - 3.0) ** 2 / 0.1)
26 calculation_3 = cp.exp(-(x - - 3.0) ** 2 / 0.5)
27
28 # Calculate RGBA values based on the density calculations
29 r = 1.0 * calculation_1 + 0.1 * calculation_2 + 0.1 * calculation_3
30 g = 1.0 * calculation_1 + 1.0 * calculation_2 + 0.1 * calculation_3
31 b = 0.1 * calculation_1 + 0.1 * calculation_2 + 1.0 * calculation_3
32 a = 0.6 * calculation_1 + 0.1 * calculation_2 + 0.01 * calculation_3
33 return r, g, b, a

18def transferFunction(x):
19 """
20 @brief Transfer function to adjust color intensity based on density values.
21 @param x: Density value for which color intensity is calculated.
22 @return Tuple of RGBA values.
23 @details Applies cupy exponential functions to simulate different material densities and calculates RGBA values.

26 # Apply exponential functions to simulate different material densities
27 calculation_1 = cp.exp(-(x - 9.0) ** 2 / 1.0)
28 calculation_2 = cp.exp(-(x - 3.0) ** 2 / 0.1)
29 calculation_3 = cp.exp(-(x - - 3.0) ** 2 / 0.5)
30
31 # Calculate RGBA values based on the density calculations
32 r = 1.0 * calculation_1 + 0.1 * calculation_2 + 0.1 * calculation_3
33 g = 1.0 * calculation_1 + 1.0 * calculation_2 + 0.1 * calculation_3
34 b = 0.1 * calculation_1 + 0.1 * calculation_2 + 1.0 * calculation_3
35 a = 0.6 * calculation_1 + 0.1 * calculation_2 + 0.01 * calculation_3
36 return r, g, b, a
